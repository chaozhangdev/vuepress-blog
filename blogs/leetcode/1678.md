```js
/**
 * @param {string} command
 * @return {string}
 */
const interpret = (command) => {
  let buffer = "";
  let result = "";
  while (command !== "") {
    if (command.slice(0, 1) === "G") {
      result += "G";
      command = command.substring(1);
    } else if (command.slice(0, 2) === "()") {
      result += "o";
      command = command.substring(2);
    } else {
      result += "al";
      command = command.substring(4);
    }
  }
  return result;
};
```
