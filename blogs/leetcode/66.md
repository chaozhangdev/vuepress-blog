```js
/**
 * @param {number[]} digits
 * @return {number[]}
 */
const plusOne = (digits) => {
  let num = BigInt(digits.join(""));
  num++;
  num = (num + "").split("");
  let result = num.map((el) => parseInt(el));
  return result;
};
```
